rating_age=toy_story_group.groupby("Rating")
rating_f=[]
age_f=[]
gender_f=[]
count_f=[]
count_fg=[]
rating_fg=[]
for group,names in rating_age:
    count=[]
    age=[]
    rating=[]
    gender=[]
    rating_g=[]
    count_g=[]
    names = names.sort_values(by=['Age'])
    for ind in names.index:
        if names["Age"][ind] not in age:
            age.append(names["Age"][ind])
            count.append(1)
            rating.append(int(group))
        else:
            ind_1=age.index(names["Age"][ind])
            count[ind_1]+=1
        if names["Gender"][ind] not in gender:
            gender.append(names["Gender"][ind])
            count_g.append(1)
            rating_g.append(int(group))
        else:
            ind_1=gender.index(names["Gender"][ind])
            count_g[ind_1]+=1
        
    count_f+=count
    age_f+=age
    rating_f+=rating
    gender_f+=gender
    count_fg+=count_g
    rating_fg+=rating_g
